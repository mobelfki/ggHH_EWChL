\title{Probing the trilinear Higgs boson coupling in di-Higgs production at NLO QCD with full $m_t$-dependence}

\author[a]{G.~Heinrich,}

\preprint{{\small  MPP-2018-xxx}}

\abstract{
 We present  NLO QCD results for Higgs boson pair 
  production with the trilinear Higgs boson coupling  left as a free
  parameter, and with full top quark mass dependence.
  We provide differential results at 14\,TeV and 27\,TeV and discuss
  the implications of anomalous trilinear couplings.
 The matrix elements have been implemented in the {\tt POWHEG-BOX-V2}
  Monte Carlo framework and the corresponding generator is publicly available.
}
 

\keywords{Higgs phenomenology, NLO QCD, BSM, future colliders}
